<body>

    <h1>Calculadora de Vendas</h1>

    <label>Investimento Mensal: <input type="number" id="investimento" placeholder="R$"></label>

    <label>Preço por Lead: <input type="number" id="precoLead" placeholder="R$"></label>

    <label>Taxa de Agendamento (%): <input type="number" id="taxaAgendamento" placeholder="%"></label>

    <label>Taxa de Comparecimento (%): <input type="number" id="taxaComparecimento" placeholder="%"></label>

    <label>Taxa de Conversão (%): <input type="number" id="taxaConversao" placeholder="%"></label>

    <label>Ticket Médio: <input type="number" id="ticketMedio" placeholder="R$"></label>

    <button onclick="calcular()">Calcular</button>

 

    <h2>Resultados</h2>

    <p>Número de Leads: <span id="numLeads"></span></p>

    <p>Faturamento: <span id="faturamento"></span></p>

    <p>CAC: <span id="cac"></span></p>

    <p>ROAS: <span id="roas"></span></p>

 

    <script>

        function calcular() {

            const investimento = parseFloat(document.getElementById('investimento').value);

            const precoLead = parseFloat(document.getElementById('precoLead').value);

            const taxaAgendamento = parseFloat(document.getElementById('taxaAgendamento').value) / 100;

            const taxaComparecimento = parseFloat(document.getElementById('taxaComparecimento').value) / 100;

            const taxaConversao = parseFloat(document.getElementById('taxaConversao').value) / 100;

            const ticketMedio = parseFloat(document.getElementById('ticketMedio').value);

 

            const numLeads = investimento / precoLead;

            const numAgendamentos = numLeads * taxaAgendamento;

            const numComparecimentos = numAgendamentos * taxaComparecimento;

            const numVendas = numComparecimentos * taxaConversao;

            const faturamento = numVendas * ticketMedio;

            const cac = investimento / numVendas;

            const roas = faturamento / investimento;

 

            document.getElementById('numLeads').innerText = numLeads.toFixed(2);

            document.getElementById('faturamento').innerText = `R$ ${faturamento.toFixed(2)}`;

            document.getElementById('cac').innerText = `R$ ${cac.toFixed(2)}`;

            document.getElementById('roas').innerText = roas.toFixed(2);

        }

    </script>
